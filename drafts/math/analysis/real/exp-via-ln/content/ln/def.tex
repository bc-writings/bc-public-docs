\begin{defi}
	Le \focus{logarithme népérien} est la fonction $\ln$ définie sur $\RRsp$ par $\ln x = \integrate*{\frac1t}{t}{1}{x}$.
\end{defi}


% ----------------------- %


\begin{fact}
	$\forall x \in \RRsp$,
	$\sder{\ln}{1} x = \frac1x$.
	%
	En particulier,
	la fonction $\ln$ est strictement croissante sur $\RRsp$,
	et
	sa représentation graphique n'admet aucune tangente horizontale.
\end{fact}
