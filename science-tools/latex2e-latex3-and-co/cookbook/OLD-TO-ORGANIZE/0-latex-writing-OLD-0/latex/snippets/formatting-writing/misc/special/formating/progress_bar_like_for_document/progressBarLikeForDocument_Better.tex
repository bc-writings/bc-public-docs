	\usepackage{lipsum,fancyhdr}
	\pagestyle{fancy}

	\usepackage[lastpage]{zref}

	\pagenumbering{Roman}

	\makeatletter
		\zref@newprop*{numpage}{\the\value{page}}
		\zref@addprop{main}{numpage}
		\lhead{\rule{\dimexpr \textwidth * \the\value{page}/\zref@extractdefault{LastPage}{numpage}{1}}{2mm}}
	\makeatother
