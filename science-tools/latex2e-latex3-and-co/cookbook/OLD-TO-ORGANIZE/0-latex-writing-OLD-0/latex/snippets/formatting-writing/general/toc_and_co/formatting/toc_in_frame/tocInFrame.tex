\documentclass{article}
	\usepackage{framed}
	\usepackage{tikz}

	\newcommand{\tkzroundedframecmd}[1]{%
		\begin{tikzpicture}
			\node[fill=orange!10 , draw=blue!50!black , rounded corners , thick]{
				\begin{minipage}{\textwidth}
					#1
				\end{minipage}
			};
		\end{tikzpicture}
	}

	\newenvironment{tkzroundedframe}{%
		\def\FrameCommand{\tkzroundedframecmd}%
			\MakeFramed{\hsize=.8\textwidth\FrameRestore}}%
		{\endMakeFramed}

% Redéfinition de la table des matières
	\let\oldtableofcontents\tableofcontents
	\renewcommand{\tableofcontents}{%
		\begin{tkzroundedframe}
			\oldtableofcontents
		\end{tkzroundedframe}
	}
