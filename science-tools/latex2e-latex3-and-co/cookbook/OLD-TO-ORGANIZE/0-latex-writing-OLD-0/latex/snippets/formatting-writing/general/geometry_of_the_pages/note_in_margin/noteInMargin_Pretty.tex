\documentclass{article}
	\usepackage{lipsum}
	\usepackage{xcolor}
	\usepackage{mdframed}

	\setlength{\textwidth}{4.0in}%
	\setlength{\marginparwidth}{2.0in}%

	\mdfdefinestyle{MyMarginNoteStyle}{
		topline=false,
		bottomline=false,
		rightline=false,
		backgroundcolor=yellow!20
	}

	\newcommand{\MarginNote}[1]{%
		\marginpar{%
			\begin{mdframed}[style=MyMarginNoteStyle]%
				\textbf{Notes}%
				\par\bigskip%
				#1%
			\end{mdframed}%
		}%
	}%


\begin{document}

\lipsum[1-3]

\MarginNote{\lipsum[1]}

\lipsum[4-6]

\MarginNote{\lipsum[2]}

\end{document}
