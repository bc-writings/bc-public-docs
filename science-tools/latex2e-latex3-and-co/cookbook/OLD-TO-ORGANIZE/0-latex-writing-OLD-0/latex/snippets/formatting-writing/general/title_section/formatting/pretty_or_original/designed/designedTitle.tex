\documentclass{article}
	\usepackage[utf8]{inputenc}
	\usepackage[T1]{fontenc}
	\usepackage{lmodern}

	\usepackage{tikz}
	\usetikzlibrary{decorations.text}
	\usetikzlibrary{fadings}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%  Titre de chapitres en forme de coquillage  %%
%%  #1 = le numéro ; #2 = sous-titre ;         %%
%%  #3 = titre ; #4 = couleur                  %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

	\newenvironment{ant}{\fontfamily{anttlc}\selectfont}{}
	
	\newsavebox{\chaptertwo}

	\newenvironment{chap2}[4]{
		\noindent
		\begin{tikzfadingfrompicture}[name=chaptertwotitle]
			\node[text=transparent!20]{
				\fontfamily{anttlc}\selectfont\Huge\bfseries #3
			};
		\end{tikzfadingfrompicture}
		%
		\begin{lrbox}{\chaptertwo}
			\begin{minipage}{0.8\textwidth}
				\begin{tikzpicture}
					\node[inner sep=0cm] (textbox){
						\phantom{\fontfamily{anttlc}\selectfont\Huge\bfseries #3}
					};
					\shadedraw[
						path fading=chaptertwotitle,
						fit fading=false,
						bottom color=white,
						top color=blue,
						right
					] (textbox.south west) rectangle (textbox.north east);
				\end{tikzpicture}
			\end{minipage}
		\end{lrbox}
		%
		\begin{tikzpicture}
			\foreach \i in {0,5,10,...,350}{
				\pgfmathparse{\i*10/36}
				\let \p \pgfmathresult

				\pgfmathparse{100-\p}
				\let \c \pgfmathresult

				\pgfmathparse{1.25*\c/100}
				\let \G \pgfmathresult

				\pgfmathparse{0.62*\c/100}
				\let \P \pgfmathresult

				\shade[
					inner color=#4!\p,
					outer color=#4!\c,
					rotate=\i
				] (0,0) ellipse (\G cm and \P cm);
			}

			\draw[
				decorate,
				decoration = {
					text along path,text={|\bf\color{black}|Chapitre}
				}
			] (-1.04,-.02) arc (180:280:1cm and .75cm);

			\draw[
				decorate,
				decoration = {
					text along path,
					text = {|\bf\color{white}|Chapitre}
				}
			] (-1,0) arc (180:280:1cm and .75cm);

			\node[left] at (1.27,-0.02) {
				\begin{large}%
					\begin{ant}%
						\textbf{\textcolor{black}{#1}}%
					\end{ant}%
				\end{large}
			};

			\node[left] at (1.25,0) {
				\begin{large}%
					\begin{ant}%
						\textbf{\textcolor{white}{#1}}%
					\end{ant}%
				\end{large}
			};

			\node[right,color=black] at (1.52,0.5) {
				\begin{Huge}%
					\begin{ant}%
						\textbf{#3}%
					\end{ant}%
				\end{Huge}
			};

			\node[right,color=#4] at (1.48,.52) {
				\usebox{\chaptertwo}
			};

			\node[right,color=#4!50] at (1.6,-0.25) {
				\begin{scriptsize}%
					\begin{ant}%
						\textit{#2}%
					\end{ant}%
				\end{scriptsize}
			};

		\end{tikzpicture}
	}
