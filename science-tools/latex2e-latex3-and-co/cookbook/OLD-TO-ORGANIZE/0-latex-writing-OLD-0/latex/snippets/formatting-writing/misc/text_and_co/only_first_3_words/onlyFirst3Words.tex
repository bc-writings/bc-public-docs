\documentclass{article}
% I need \ifblank to put \dots
% if something nonblank gets deleted
	\usepackage{etoolbox} 

	\makeatletter
		\def\@@threefirstwords #1 #2 #3 #4\\#5{%
			#1 #2 #3\ifblank{#4}{}{#5}%
		}

		\edef\@threefirstwords#1#2{%
			\noexpand\@@threefirstwords #1\space\space\space\noexpand\\{#2}%
		}

		\newcommand{\threefirstwords}[1]{\@threefirstwords{#1}{\dots}}
	\makeatother

	\newcommand{\test}[1]{%
		\noindent #1 \\ \threefirstwords{#1} \\%
	}


\begin{document}

\test{This is a test to see the efectcs.}


\test{This is another small test}


\test{Nothing more.}


\test{Is it finished ?}


\end{document}
