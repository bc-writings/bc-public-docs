\documentclass{article}
	\usepackage{amsmath}% http://ctan.org/pkg/amsmath


\begin{document}

\begin{align*}
	\left.\begin{array}{r@{\mskip\thickmuskip}l}
		\nabla f(x) -\mu X^{-1} e + A(x) \lambda - z &= 0, \\
		c(x) &= 0, \\
		-\mu X^{-1} e + z &= 0, \\
		x &> 0 \\
		z &> 0
	\end{array} \right\}
	\quad \implies \quad
	\left\{\begin{array}{r@{\mskip\thickmuskip}l}
		\nabla f(x) + A(x) \lambda - 2z &= 0, \\
		c(x) &= 0, \\
		-\mu X^{-1} e + z &= 0, \\
		x &> 0 \\
		z &> 0
	\end{array}\right.
\end{align*}
