\documentclass[10pt,a4paper]{article}
	\usepackage[utf8x]{inputenc}
	\usepackage{amsmath,amssymb}
	\usepackage{array}

	\newcommand{\function}[5]{ %
		\begin{array}{l@{\;}|@{\;}r@{\;}c@{\;}l}
				#1 \colon & #2 & \longrightarrow & #3 \\
								 & #4 & \longmapsto & #5 %
		 \end{array}}

\begin{document}
	Soit $\function{f}{E}{F}{x}{x^3}$

	Ou bien,
	 \[ \function{f}{E}{F}{x}{x^3} \]
\end{document}
