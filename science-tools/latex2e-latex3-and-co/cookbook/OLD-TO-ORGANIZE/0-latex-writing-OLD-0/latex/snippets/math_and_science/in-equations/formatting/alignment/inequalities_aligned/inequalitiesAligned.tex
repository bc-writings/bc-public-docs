$\begin{alignedat}{3}
	-1             & \leq (-1)^n                     && \leq 1\\
	-1+n           & \leq (-1)^n+n                   && \leq 1+n\\
	-1+2           & \leq (-1)^n+2                   && \leq 1+2\\
	\dfrac{1}{3}   & \leq \dfrac{1}{(-1)^n+2}        && \leq 1\\
	\dfrac{n-1}{3} & \leq \dfrac{(-1)^n+n}{(-1)^n+2} && \leq 1+n\\
\end{alignedat}$
