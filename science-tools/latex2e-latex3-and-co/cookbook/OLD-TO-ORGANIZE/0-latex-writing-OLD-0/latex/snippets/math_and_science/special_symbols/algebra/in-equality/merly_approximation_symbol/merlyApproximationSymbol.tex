\documentclass{article}
	\usepackage{amsmath}

	\def\app#1#2{%
		\mathrel{%
			\setbox0=\hbox{$#1\sim$}%
			\setbox2=\hbox{%
				\rlap{\hbox{$#1\propto$}}%
				\lower1.1\ht0\box0%
			}%
			\raise0.25\ht2\box2%
		}%
	}
	\def\approxprop{\mathpalette\app\relax}


\begin{document}

\[ a \approxprop b \]

\centerline{$a \approxprop b$}

\[ \sum_{a \approxprop b} \]

\[ X_{Y_{a \approxprop b}} \]
\end{document}
