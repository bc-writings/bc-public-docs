		\node[]  at (\x,0)   {$\ytext$};

		\node[] (y_\x) at (\x,-0.25) {};%
	}
	\draw[->,red,thin,>=stealth'] (x_1) edge [bend left] node[above]{$\times 5$} (x_4);

	\draw[color=blue,->,thin,>=stealth'] (4.5,.65) edge [distance=1.2cm,bend left=80 ] node [above,sloped,midway]{$\times 4$} (4.5,-.15);
