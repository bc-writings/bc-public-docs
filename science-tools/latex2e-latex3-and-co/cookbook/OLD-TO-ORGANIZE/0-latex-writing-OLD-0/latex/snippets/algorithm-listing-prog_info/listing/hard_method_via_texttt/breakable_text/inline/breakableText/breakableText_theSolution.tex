	\makeatletter
		\newcommand*\breakabletexttt{%
			\begingroup
			\@makeother\_% This allows the use of special
			\@makeother\#% character _ et # ...
			\ttfamily
			\afterassignment\@breakabletexttt
			\edef\temp@%
		}

		\newcommand*\@breakabletexttt{%
			\expandafter\breakable@texttt\temp@\breakable@texttt%
		}

		\newcommand*\breakable@texttt[1]{%
			\ifx\breakable@texttt#1\endgroup
			\else
				#1\hskip\z@ plus.1pt minus.1pt
				\expandafter\breakable@texttt
			\fi%
		}
	\makeatother
