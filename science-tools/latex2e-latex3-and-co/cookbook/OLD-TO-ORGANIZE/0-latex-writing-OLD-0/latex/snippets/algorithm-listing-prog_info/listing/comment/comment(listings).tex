	\usepackage{xcolor}
	\usepackage{listings}

	\lstset{
		language=[LaTeX]tex,
		morecomment=[n][\color{blue!50!black}\bfseries]{\{}{\}},
		morecomment=[s][\color{green!50!black}\bfseries\slshape]{$}{$},
	}


\begin{document}

\begin{lstlisting}

\end{document}
\end{lstlisting}

\end{document}
