\documentclass[10pt,a4paper]{article}
	\usepackage[utf8x]{inputenc}
	\usepackage{ucs}
	\usepackage{xstring}

	\newcommand{\key}[1]{
		\fbox{#1\strut}
	}

	\makeatletter
		\def\keySplit#1{\key@Split#1\@nil{}}
		\def\key@Split#1#2\@nil#3{%
			\ifx\@empty#2\@empty %
				\expandafter\@firstoftwo
			\else %
				\expandafter\@secondoftwo %
			\fi
			{#3\key{#1}}
			{\key@Split#2\@nil{#3\key{#1}}}}
	\makeatother


\begin{document}

\key{5} \key{$\times$} \key{$x$} \key{$x$}
and
\key{ArcTan}


\keySplit{5 {$\times$} {$x$} {$x$}} and \keySplit{Ar  cTan}

\end{document}
