\documentclass{article}
	\usepackage{ifmtarg}

	\makeatletter
		\newcommand{\isempty}[1]{%
			\@ifmtarg{#1}{YES}{NO}%
		}
	\makeatother

\begin{document}

\verb+\isempty{}+ : \isempty{}

\verb+\isempty{     }+ : \isempty{     }

\verb+\isempty{E}+ : \isempty{E}

\verb+\isempty{  E     }+ : \isempty{  E     }

\end{document}
