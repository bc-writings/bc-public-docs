\documentclass[11pt]{book}
	\usepackage{lstdoc,lipsum}

	\makeatletter
		\def\alist{}

		\let\sort\lst@BubbleSort 
		\def\addtolist#1#2{
			\lst@lAddTo\alist{#2}
		}

		\long\gdef\addPerson#1#2{\addtolist\alist{#1,}}

		\def\AddBio#1#2{%
			\long\expandafter\gdef\csname#1\endcsname{\textbf{#1}: #2}
			\addPerson{#1}{#2}
			\sort\alist
		}

		\def\PrintBios{%
			\@for \i:=\alist\do{%
			\csname\i\endcsname}
		}
	\makeatother


\begin{document}

\AddBio{Yiannis}{\lipsum[2]}
\AddBio{Mary}{\lipsum[3]}
\AddBio{Ann}{\lipsum[1]}

% Print
\PrintBios

\end{document}
