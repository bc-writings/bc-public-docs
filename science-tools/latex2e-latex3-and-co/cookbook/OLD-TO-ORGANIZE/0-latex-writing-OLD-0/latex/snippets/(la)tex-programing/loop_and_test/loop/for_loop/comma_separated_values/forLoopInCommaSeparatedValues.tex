\documentclass{article}
	\usepackage{xstring}
	\makeatletter
		\newcommand{\splitIt}[1]{%
			\IfSubStr{#1}{,}{\@currentVal#1\@end@token}{+#1+}%
		}
		\def\@currentVal#1,#2\@end@token{%
			+#1+\par\splitIt{#2}%
		}
	\makeatother


\begin{document}

\verb+\splitIt{a}+
\par
\splitIt{a}

\medskip

\verb+\splitIt{a,b,cggg,d,e,f,g}+
\par
\splitIt{a,b,cggg,d,e,f,g}

\end{document}
