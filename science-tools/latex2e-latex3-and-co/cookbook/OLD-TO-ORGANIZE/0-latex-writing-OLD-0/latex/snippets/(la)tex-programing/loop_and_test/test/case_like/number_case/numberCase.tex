\documentclass{article}
	\makeatletter
		\newcommand\test[1]{%
			\ifcase#1
				\or Fisrt case %
				\or Secund case %
				\or Third case %
				\else\@ctrerr %
			\fi
			%
		}
	\makeatother

\begin{document}

\verb+\test{1}+ : \test{1}

\verb+\test{2}+ : \test{2}

\verb+\test{3}+ : \test{3}

% To test error
% \test{4}    ----> LaTeX Error: Counter too large.
% \test{d}    ----> LaTeX Error: Missing number, treated as zero.

\end{document}
