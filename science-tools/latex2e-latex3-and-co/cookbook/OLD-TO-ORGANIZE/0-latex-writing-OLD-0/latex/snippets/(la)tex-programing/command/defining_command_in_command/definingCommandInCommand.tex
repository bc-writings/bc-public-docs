	\newcommand\aaa[1]{%
		\newcommand\bbb[1]{#1 ##1}%
	}


	\def\AAA#1{%
		\def\BBB##1{#1 ##1}%
	}

\begin{document}

\aaa{from aaa}
\bbb{test}

\AAA{FROM AAA}
\BBB{TEST}

\end{document}
