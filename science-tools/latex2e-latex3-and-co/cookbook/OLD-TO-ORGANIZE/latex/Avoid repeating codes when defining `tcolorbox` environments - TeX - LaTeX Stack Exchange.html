\end{document}

    tcolorbox's key-val scheme is from pgfkeys, which is auto loaded by either pgf or tikz. So you can define a style key using /.style key handler.

    \documentclass[12pt]{article}

\RequirePackage{tcolorbox}

\tcbuselibrary{minted, breakable}

\tcbset{
    doc latex common/.style={
        breakable,
        sharp corners,
        left   = 1mm, right = 1mm,
        bottom = 1mm, top   = 1mm,
    }
}

\newtcblisting{doclatex}{%
    doc latex common,
    listing side text
}

\newtcblisting{doclatex-flat}{
    doc latex common,
    bottom = 1mm, top   = 1mm,
}

\begin{document}

\begin{doclatex}
$just + A = test$
\end{doclatex}

\begin{doclatex-flat}
$just + A = test$
\end{doclatex-flat}

\end{document}
