\documentclass{article}
\usepackage{keytheorems}

%\ExplSyntaxOn
%\cs_new_eq:NN \tlIfEmptyTF \tl_if_empty:nTF
%\ExplSyntaxOff

\newkeytheoremstyle{mythmsty}
  {
    headformat=\protect\IfBlankTF{#3}{\NAME}{\NOTE},
    notebraces={}{},
    notefont=\bfseries,
    noteseparator={},
  }
\newkeytheorem{mythm}[
  numbered = false,
  name     = My theorem,
  style    = mythmsty,
]
    
\begin{document}

\begin{mythm}
   Some text
\end{mythm}

\begin{mythm}[Named theorem]
   Some text
\end{mythm}

\end{document}
