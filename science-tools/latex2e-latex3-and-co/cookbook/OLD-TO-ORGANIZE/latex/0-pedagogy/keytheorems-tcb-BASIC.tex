\documentclass{article}

\usepackage{tcolorbox}

\usepackage{keytheorems}

\tcbset{
  thmstyle/.style={
    colback  = #1!5,        
    colframe = #1!35!black,
  },
}

\newkeytheoremstyle{tcb-standard}{
  headpunct     = {},
  notebraces    = {}{}, 
  noteseparator = {: }, 
  notefont      = \bfseries, 
  bodyfont      = \normalfont,
}

\newcommand{\projetnewstyle}[1]{%
  \newkeytheoremstyle{tcb-standard-#1}{
    inherit-style = tcb-standard,
    tcolorbox     = {thmstyle=#1}
  }
}

\projetnewstyle{red}
\projetnewstyle{green}

\newkeytheorem{test}[
  name  = Test,
  style = tcb-standard-red
]

\newkeytheorem{blub}[
  style = tcb-standard-green
]


\begin{document}

\begin{test}
  Bla, blo, bli, bla, blo, bli, bla, blo, bli, bla, blo, bli, bla, blo, bli, bla...
\end{test}
\begin{blub}
  Bla, blo, bli, bla, blo, bli, bla, blo, bli, bla, blo, bli, bla, blo, bli, bla...
\end{blub}

\end{document}
