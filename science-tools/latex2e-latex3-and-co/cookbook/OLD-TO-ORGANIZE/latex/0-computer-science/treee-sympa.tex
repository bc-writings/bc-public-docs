\documentclass[tikz,border=5pt]{standalone}
% ateb: https://tex.stackexchange.com/a/713546/
\usepackage{forest}
\usetikzlibrary{arrows.meta}
\begin{document}    
\begin{forest}
  for tree={%
    edge+={-Latex},
    anchor=base,
    align=@{}c@{},
    font=\sffamily,
    s sep'+=10pt,
  },
  my label/.style={%
    if n=2{%
      label={[my label style]right:{#1}},
    }{%
      label={[my label style]left:{#1}},
    },
  },
  /tikz/my label style/.style={font=\sffamily},
  before computing xy={%
    tempdima/.max={ > OOw2+d {max x}{min x}{#1-#2} }{tree},
    for tree={%
      minimum width/.register=tempdima,
      draw,
      thin,
    },
    typeset nodes,
  },
  [{(0,0)}, my label={(0,0)}
    [{(1,1)}, my label={(0,0)}
      [{(2,2)}, my label={(0,0)}
            [{(3,3)}, my label={(0,0)}
                 [{(4,4)}, my label={(0,0)}]
            ]    
            [{(5,5)}, my label={(0,0)}]
      ]                   
    ]  
  ]
\end{forest}
\end{document}
