\documentclass{article}

\usepackage{mathtools}

\newcommand\samelenas[2]{\phantom{\text{#2}}\mathllap{\text{#1}}}

\begin{document}

$\frac{\text{test} = 0.01}
      {\text{testing} = 0.01}$

$\frac{\samelenas{test}{testing} = 0.01}
      {\text{testing} = 0.01}$

\end{document}
