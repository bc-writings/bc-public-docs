        \draw[color = red]
               ([xshift = 10pt]2-1.south west)
            |- (\myind-2.west);

        \foreach \i [evaluate = \i as \mysecond using {int(\i + \myind)}] in {1, 2} {

                |- (\mysecond-3.west);
        }
    }
