\documentclass[border={-63pt -17pt -7pt 3pt}]{standalone}

\usepackage{adigraph}

\begin{document}

%%%
% Définition de l'automate sans l'afficher.
%%%
\NewAdigraph{automata}{
  s: 0, 0 ;            % On positionne le noeud.
  t: 2, 1 : $\alpha$ ; % Choix du nom "imprimé".
}{
  s, s: 3 ; % Une boucle en s.
  s, t: 2 ; % Une connection de s vers t.
  t, s: 5 ; % Une connection de t vers s.
}

%%%
% L'affichage se fait comme suit autant de fois que l'on veut.
%%%
\automata{}

\end{document}
