\documentclass[border = 3pt]{standalone}

\usepackage[svgnames]{xcolor}
\usepackage[3d]{luadraw}

\begin{document}

\begin{luadraw}{name = vector-field-two-grids}
require 'luadraw_palettes'
require 'luadraw_shadedforms'
require 'luadraw_fields'

local f = function(x, y)
  return {1, x^2 + y^2 - 1}
end

local Nf = function(x, y)
  A = f(x, y)

  return cpx.abs(Z(A[1],A[2]))
end

local xmin, xmax, ymin, ymax = -3, 3, -3, 3

local graphview = graph:new{
  window = {xmin - 0.5, xmax + 0.5, ymin - 0.5, ymax + 0.5},
  size   = {10, 10},
  bbox   = false
}

graphview:Dshadedrectangle(   
  xmin, xmax, ymin, ymax, 
  getPal("Jet"),
  {
    values     = Nf,
    bar        = "right"
  }
)

graphview:Show()
\end{luadraw}

\end{document}
