\usepackage{multicol}

\newcommand\example{
    $\interval{a, b}$
    \quad
    $\interval[big]{a, b}$

    $\interval{\dfrac{1}{2}, \dfrac{2}{3}}$
    \quad
    $\interval[Big]{\dfrac{1}{2}, \dfrac{2}{3}}$
    \quad
    $\interval[auto]{\dfrac{1}{2}, \dfrac{2}{3}}$

    $\interval[c]{a, b}$
    \quad
    $\interval[oc]{-oo, b}$
    \quad
    $\interval[o]{a, oo}$
    \quad
    $\interval{a, +oo}$

    $\interval{-oo, oo}$
    \quad
    $\interval{-oo, +oo}$

    $\interval{1, 2} \cup \interval[co]{2, 3} = \interval{1, 3}$

    $\interval[c, sep=;]{a, b}$
    \quad
    $\interval[oc, sep=..]{a, b}$
    \quad
    $\interval[oc, sep=..]{-oo, +oo}$
}
