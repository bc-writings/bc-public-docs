\section*{Standard}
Standard typesetting of intervals

$\interval{a,b}$\quad
$\interval[big]{a,b}$\quad
$\interval[auto]{\dfrac{1}{2},\dfrac{2}{3}}$

$\interval[c]{a,b}$

$\interval[cc,sep=..]{a,b}$

$\interval[oc]{-,b}$

$\interval[oo]{a,-}$\qquad
$\interval[oo]{a,+}$

$\interval[o]{-,-}$\qquad
$\interval[o]{-,+}$

$\interval{1,2}\cup\interval[co]{2,3}=\interval[oo]{1,3}$
