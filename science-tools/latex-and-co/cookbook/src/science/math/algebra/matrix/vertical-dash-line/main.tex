\documentclass[border = 3pt]{standalone}

%%%
% Le \pack ''nicematrix'' est le couteau suisse des matrices.
%%%
\usepackage{nicematrix}

%%%
% Le \pack ''tikz'' est nécessaire pour décorer les matrices.
%%%
\usepackage{tikz}

\begin{document}

%%%
% Réglages "maison" pour tracer nos délimitations hachurées.
%%%
\NiceMatrixOptions{
  custom-line = {
% Lignes hachurées horizontales.
    command     = \hdashedline,
% Lignes hachurées verticales.
    letter      = I ,
    tikz        = dashed ,
    total-width = \pgflinewidth
  }
}

$H
=
 \begin{pNiceArray}{ccIcc}[
   first-row,
   first-col
 ]
     & a & b & c & d  \\
   e & 1 & 2 & 3 & 4  \\
   f & 5 & 6 & 7 & 8  \\
 \hdashedline
   g & 9 & 0 & 1 & 2  \\
   h & 3 & 4 & 5 & 7
 \end{pNiceArray}
$

\end{document}
