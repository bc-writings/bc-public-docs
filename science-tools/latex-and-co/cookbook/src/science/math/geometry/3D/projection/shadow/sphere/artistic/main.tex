\documentclass[border = 3pt]{standalone}

\usepackage[svgnames]{xcolor}
\usepackage[3d]{luadraw}

\begin{document}

\begin{luadraw}{name = central-proj-of-sphere-artistic}
local r = 2
local C = M(-1, -1, 0)

local z0    = -3
local light = M(-5, -5, 5)

local proj = function(A)
  return proj3dO(
    A,
    {z0*vecK, vecK},
    A - light
  )
end

local graphview = graph3d:new{
  window3d = {-5, 8, -5, 8, -3, 5},
  adjust2d = true,
  bbox     = false,
  viewdir  = perspective("yz", 0.35, 60),
  size     = {10, 10}
}

graphview:Linewidth(12)

local S = {C, r}

local I = (light + C) / 2
local R = pt3d.abs(C - light) / 2

local C_t, r_t, normal_t = interSS(S, {I, R})

local shadow = ftransform3d(
  circle3d(C_t, r_t, normal_t)[1],
  proj
)

local beam = pyramid(shadow, light, true)

graphview:Dfacet(
  graphview:Plane2facet({z0*vecK, vecK}),
  {
    color    = "lightgray",
    contrast = 0
  }
)

graphview:Dseg3d({M(-5, -5, z0), light})

graphview:Dballdots3d(light, "Yellow", 2)

graphview:Dpolyline3d(
  shadow,
  true,
  "draw = none, fill = DarkGray, fill opacity = 0.8"
)

graphview:Dsphere(
  C, r,
  {
    color = "Cyan",
    mode  = mBorder
  }
)

graphview:Dpoly(
  beam,
  {
    color    = "yellow",
    opacity  = 0.4,
    backcull = true,
    contrast = 0.5,
    mode     = mShadedOnly
  }
)

graphview:Show()
\end{luadraw}

\end{document}
