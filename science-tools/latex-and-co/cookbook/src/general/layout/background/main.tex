\documentclass[a4paper]{article}

%%%
% Nous avons besoin de produire du faux contenu.
%%%
\usepackage{lipsum}

%%%
% La bibliothèque ''luadraw'' allie une facilité d’utilisation
% à un rendu particulièrement soigné.
%%%
\usepackage{luadraw}

%%%
% Définition d'une fonction ¨lua pour dessiner, via ¨luadraw,
% la même suite de cercles au fond de chaque page.
%%%
\begin{luacode*}
function background()
------
-- ¨Def de la zone graphique.
------
  local graphview = graph:new{
    margins = {0, 0, 0, 0},
    size    = {18, 18}
  }

------
-- Réglages liés aux lignes.
------
  graphview:Linecolor("lightgray")

------
-- Une suite de cercles.
------
  for j = 1, 5 do
    for k = 1, 10*j do
      graphview:Dcircle(
        Zp(j, k * math.pi / (5*j)),
        1
      )
    end
  end

------
-- On renvoie directement le code ¨tikz.
------
  graphview:Sendtotex()
end
\end{luacode*}

%%%
% Création d'une macro sans argument qui sera utilisée lors
% de la production du fond de chaque page.
%%%
\newcommand\background{\directlua{background()}}

%%%
% Utilisation d'un point d'ancrage du noyau ¨latex pour agir
% lors de la fabrication des fonds des pages.
%%%
\AddToHook{shipout/background}{%
    \put(\paperwidth/2 - 9cm, -\paperheight/2 - 9cm){\background}%
}

\lipsum[1-2]

\newpage

\lipsum[1-2]

\end{document}
