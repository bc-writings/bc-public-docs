\section*{\null\hfill La clé \texttt{tabularnote} \hfill\null}

\begin{table}[ht!]
  \centering
  \caption{Annoter un tableau}

  \begin{NiceTabular}{c|c|c}[
    tabularnote = {Une note pour le tableau.}
  ]
    \toprule

    Catégorie & Coût & Prix \\

    \midrule

    A & 200 & 100 \\
    B & 100 \tabularnote{Une note pour un prix.}
            & 200 \\
    C & 300 & 300 \\

    \bottomrule
  \end{NiceTabular}
\end{table}
