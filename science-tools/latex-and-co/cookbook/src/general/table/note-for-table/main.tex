\section{Via une clé pour le tableau lui-même}

\begin{table}[ht!]
  \centering
  \caption{Annoter un tableau}

  \begin{NiceTabular}{c|c|c}[
    tabularnote = {Une note pour le tableau.}
  ]
    \toprule

    Category & Cost & Value \\

    \midrule

    Info. A & 200 & 200 \\
    Info. B & 100 \tabularnote{Une note pour un nombre.}
                  & 200 \\
    Info. C & 300 & 100 \\

    \bottomrule
  \end{NiceTabular}
\end{table}
