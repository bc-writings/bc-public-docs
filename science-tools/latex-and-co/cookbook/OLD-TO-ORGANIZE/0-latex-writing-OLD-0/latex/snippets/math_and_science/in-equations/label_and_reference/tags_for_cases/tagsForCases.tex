\documentclass[10pt,a4paper]{article}
	\usepackage{ucs}
	\usepackage{empheq}
	\newenvironment{tagcases}[1][]
		{\empheq[left={#1\empheqlbrace}]{alignat=2}}
		{\endempheq}


\begin{document}

\begin{tagcases}[A=]
	& \frac{r}{a} &&\qquad  r > a \tag{G} \\
	& \frac{a}{r} &&\qquad  r < a \tag{H}
\end{tagcases}

\end{document}
