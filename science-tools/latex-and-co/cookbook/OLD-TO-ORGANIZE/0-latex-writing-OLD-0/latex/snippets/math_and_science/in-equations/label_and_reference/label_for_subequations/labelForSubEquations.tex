\documentclass{article}
	\usepackage{amsmath}
	\usepackage{cleveref}

	\crefname{pluralequation}{eqs.}{eqs.}
	\Crefname{pluralequation}{Eqs.}{Eqs.}

	\crefformat{equation}{#2eq.~(#1)#3}
	\Crefformat{equation}{#2Eq.~(#1)#3}


\begin{document}

One small example...
\begin{subequations} \label{parenteq} \label[pluralequation]{parenteqs}
	\begin{align}
		x + y = z \label{subeq1}
		\\
		a + b = c \label{subeq2}
	\end{align}
\end{subequations}

\Cref{parenteqs} consist of \cref{subeq1} and \cref{subeq2}.

\end{document}
