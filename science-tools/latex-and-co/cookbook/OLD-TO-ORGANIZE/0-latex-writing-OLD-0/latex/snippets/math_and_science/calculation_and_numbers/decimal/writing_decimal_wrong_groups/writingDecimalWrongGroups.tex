\documentclass[10pt, a4paper]{article}
	\usepackage{xstring}
	
	\newcommand\ugly[2][\,]{ %
		\begingroup %
			\noexpandarg\StrSubstitute{#2}{.}{#1} %
		\endgroup %
	}

\begin{document}
	Un nombre horrible : $\ugly{1234.56.78.90}$

	Encore pire : $\ugly[\quad]{1234.56.78.90}$
\end{document}
