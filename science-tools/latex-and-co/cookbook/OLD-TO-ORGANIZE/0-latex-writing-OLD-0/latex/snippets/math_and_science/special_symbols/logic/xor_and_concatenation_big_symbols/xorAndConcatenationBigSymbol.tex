\documentclass[10pt,a4paper]{article}
	\usepackage[utf8x]{inputenc}
	\usepackage{ucs}
	\usepackage{amsmath}
	\usepackage{amsfonts}
	\usepackage{amssymb}

	\newcommand{\bigxor}{%
		\mathop{%
			\mathchoice
			{\textstyle\bigoplus}{\textstyle\bigoplus}
			{\scriptstyle\bigoplus}{\scriptscriptstyle\bigoplus}%
		}%
	}

	\newcommand{\bigconc}{\mathop{\mathpalette\bigconcinn\relax}}
	\newcommand{\bigconcinn}[2]{%
		\vcenter{\hbox{$\bigconcchoose#1\bigconcsize|\mkern1mu\bigconcsize|$}}%
	}
	\newcommand{\bigconcchoose}[1]{%
		\def\bigconcsize{}%
		\ifx#1\displaystyle
			\let\bigconcsize\Big
		\else
			\ifx#1\textstyle
				\let\bigconcsize\big
			\fi
		\fi#1%
	}


\begin{document}

\[ \bigxor_{i=0}^{3} X_{i} \]

\[ \bigconc_{i=0}^{3} X_{i} \]

\end{document}
