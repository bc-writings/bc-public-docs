\documentclass[12pt]{article}
	\usepackage{graphicx}

	\makeatletter
		\def\underparen#1{%
		\setbox0=\hbox{#1}%
		\stackrel{\box0}{\kern-\dp0\resizebox{\wd0}{\ht0}{\rotatebox{90}{(}}}}
	\makeatother


\begin{document}

$\underparen{The quick brown fox jumped over the lazy dog}$

\end{document}
