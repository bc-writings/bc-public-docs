\begin{align*}
  e^x &=\sum_{n=0}^\infty \frac{x^n}{n!}\\
  e^x+2e^{-x} &=\sum_{n=0}^\infty \frac{x^n}{n!}+2\sum_{n=0}^\infty\frac{(-x)^n}{n!}
  \boxed{=\sum_{n=0}^\infty \frac{x^n}{n!}\left(1+2(-1)^n\right)}
\end{align*}
