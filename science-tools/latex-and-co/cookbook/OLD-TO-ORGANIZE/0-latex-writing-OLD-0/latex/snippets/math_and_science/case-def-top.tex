            0, \\

        \end{cases}&& \def\arraystretch{1.2}\begin{array}{@{}l}
        t\in[t_0, t_1)\\
        t\in[t_1, t_f],
        \end{array}
         0 < c < 1\\
    u^*(t) &= \begin{cases}
            0, \\
            1, 
    \end{cases} &&\def\arraystretch{1.2}\begin{array}{@{}l}
     t\in[t_0, t_2)\\

    \end{array}
     -1 < c < 0\\
    u^*(t) &= 1, &&t\in[t_0, t_f], c \leq -1\\
    u^*(t) &= 0, &&t\in[t_0, t_f], c = 0
\end{alignat*}
