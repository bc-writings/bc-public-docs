\begin{pspicture}(-3,-3)(3.5,2.5)
	\psset{Decran=20,RotZ=-160}
	\psSolid[
		object=tore,
		fcol = 0 256 768 {
			/i ED
			i 8 i 32 add {
				dup 3 add 1 exch {
					32 i 127 add { (Blue) } for
				} for
			} for
		} for
		132 256 900 {
			/i ED
			i 8 i 32 add {
				dup 3 add 1 exch {
					32 i 127 add { (Red) } for
				} for
			} for
		} for,
		r1 = 2.5, r0 = 1.5, ngrid =32 32, %numfaces = all, fontsize = 5,
		linewidth=0.1pt
	]
\end{pspicture}
