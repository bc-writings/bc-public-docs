    \documentclass{article}

    \usepackage{txfonts}
    \usepackage{xcolor}

    \makeatletter
    \newcommand\breakabletexttt[1]{\texttt{\breakable@texttt#1\@nil}}
    \def\@gobble@fi#1\fi{\fi#1}
    \def\breakable@texttt#1#2\@nil{%
      \ifcat\expandafter\noexpand#1\relax
        \expandafter#1%
      \else
        \@gobble@fi#1\hspace{0pt plus 0.1pt minus 0.1pt}%
      \fi
      \ifx\relax#2\relax
        %
      \else
        \@gobble@fi\breakable@texttt#2\@nil
      \fi
    }
    \makeatother
