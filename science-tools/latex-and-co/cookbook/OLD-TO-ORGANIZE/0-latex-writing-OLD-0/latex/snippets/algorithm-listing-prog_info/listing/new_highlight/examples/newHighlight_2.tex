%	* Lettres i et I
		{î}{{\^i}}1 {ï}{{\"i}}1
		{Î}{{\^I}}1 {Ï}{{\"I}}1

		{ô}{{\^o}}1
		{Ô}{{\^O}}1

	\lstdefinestyle{csscode}{
		language=acsl,
		comment={[s]{/*}{*/}},
		keywordstyle=[1]\color{blue},
		keywordstyle=[2]\color{orange!70!black},

			text-align,

		keywords=[2]{

		}
	}

{\lstset{style=csscode,title=CSS,#1}}
