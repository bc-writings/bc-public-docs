\documentclass{book}
	\newcommand*\cleartoleftpage{%
		\clearpage
		\ifodd\value{page}\hbox{}\newpage\fi
	}


\begin{document}

Title on right side

\cleardoublepage
Other title on right side

\cleartoleftpage
On the left side

\cleartoleftpage
Also on the left side

\cleardoublepage
Right side again

\end{document}
