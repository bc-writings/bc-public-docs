\documentclass{article}
	\usepackage{titlesec}
	\usepackage{titletoc}

	\renewcommand{\thesection}{\Roman{section}}
	\titlespacing*{\section}
		{1cm}% decalage a gauche
		{3.5ex plus 1ex minus .2ex}% espacement vertical avant
		{2.3ex plus .2ex}% espacement vertical apres

	\titlecontents{section}%
		[0em]% decalage a gauche
		{\addvspace{1em plus 0pt}\bfseries}% code avant
		{\thecontentslabel.~}% code lorsque qu'il y a un numero
		{}% code lorsque qu'il n'y a pas un numero
		{\hfill\contentspage}% code pour le numero de page
		[\addvspace{0pt}]% code apres
