\documentclass{article}
	\usepackage{xcolor}

	\pagestyle{empty}

	\newsavebox{\mytablebox}
	\setlength{\abovecaptionskip}{0pt}
	\newenvironment{mytable}[2]{%
		\begin{table}%
		\begin{lrbox}{\mytablebox}%
		\begin{minipage}{#2}%
		\caption{#1}%
	}{
		\end{minipage}%
		\end{lrbox}%
		\colorbox{orange}{\usebox{\mytablebox}}
		\end{table}%
	}
