		\vspace*{2ex}%
		\noindent
		\boxput*(-0.9,1){\makebox[\poscadre][l]{\textwidth-2\fboxsep}\usebox\titreBoite}%
		{\fboxsep6pt%
		\shadowbox{\parbox{\textwidth-2\fboxsep-2\fboxrule-\shadowsize}{\vspace{1.5ex}\textsl{#3}}}}%
	}
