\documentclass{article}
	\usepackage{multicol}
	\usepackage{enumitem}
	
	\makeatletter
		\def\enit@endenumerate{%
			\enit@after
			\endlist

			\enit@toks\expandafter{\enit@savekeys}%

			\xdef\enit@afterlist{%
				\def\expandafter\noexpand
					\csname enit@savekeys@\@currenvir\endcsname{\the\enit@toks}%
				\global\def\expandafter\noexpand
					\csname enit@resume@\@currenvir\endcsname{%
						\csname c@\@listctr\endcsname
						\the\csname c@\@listctr\endcsname
					}
			}%
			\aftergroup\enit@afterlist}
	\makeatother

	\setenumerate{itemsep=6pt} 


\begin{document}

\begin{multicols}{2}
	\begin{enumerate}
		\item a : Start of multicols
		\item b
		\item c
		\item d : End of multicols
	\end{enumerate}
\end{multicols}

\begin{enumerate}[resume]
	\item E
	\item F
\end{enumerate}

\begin{multicols}{2}
	\begin{enumerate}[resume]
		\item g : Start of multicols
		\item h
		\item i : End of multicols
	\end{enumerate}
\end{multicols}

\end{document}
