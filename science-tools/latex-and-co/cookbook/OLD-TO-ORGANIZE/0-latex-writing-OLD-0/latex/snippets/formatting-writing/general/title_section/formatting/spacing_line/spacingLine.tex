\documentclass{article}
	\usepackage{titlesec}

	\setcounter{secnumdepth}{1}% enlève la numérotation après les sections
	\titleformat{\subsubsection}
		{\normalfont\fontsize{11pt}{13pt}\selectfont\bfseries}% apparence commune au titre et au numéro
		{\thesubsubsection}% apparence du numéro
		{1em}% espacement numéro/texte
		{}% apparence du titre


\begin{document}

\section{Titre}

\subsection{Titre}

\subsubsection{Titre}

\paragraph{Titre}

\end{document}
