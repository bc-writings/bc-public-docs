    \documentclass{article}

    \usepackage[frenchb]{babel}

    \newcounter{question}
    \newcounter{sousquestion}[question]
    \renewcommand{\thequestion}{\arabic{question}}
    \renewcommand{\thesousquestion}{\thequestion.\alph{sousquestion}}
    \makeatletter
    \newif\if@following@question
    \def\question{%
      \ifhmode\par\fi
      \refstepcounter{question}%
      \@following@questiontrue
      \everypar{\hspace{-\parindent}%
                \@hangfrom{\quad\thequestion.\enspace}%
                \everypar{}%
               \@following@questionfalse}%
    }
    \def\subquestion{%
      \ifhmode\par\fi
      \refstepcounter{sousquestion}%
      \everypar{\hspace{-\parindent}%
                \@hangfrom{\if@following@question
                             \quad\thequestion.\enspace
                           \else
                             \phantom{\quad\thequestion.\enspace}%
                           \fi
                           (\alph{sousquestion})\enspace
                          }%
                \everypar{}%
               \@following@questionfalse}
    }
    \makeatother
