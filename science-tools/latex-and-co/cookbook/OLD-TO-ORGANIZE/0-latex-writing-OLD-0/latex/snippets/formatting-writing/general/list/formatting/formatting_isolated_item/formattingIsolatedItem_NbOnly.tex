\documentclass{article}
	\usepackage{xcolor}


\begin{document}

\begin{enumerate}
	\item First item.
	{
		\renewcommand\labelenumi{\fbox{\theenumi}}
		\item Second boxed item.
	}
	\item Third item.
\end{enumerate}

\begin{enumerate}
	\item First item.
	{
		\renewcommand\labelenumi{\colorbox{red}{\theenumi}}
		\item Second boxed and colored item.
	}
	\item Third item.
\end{enumerate}

\end{document}
