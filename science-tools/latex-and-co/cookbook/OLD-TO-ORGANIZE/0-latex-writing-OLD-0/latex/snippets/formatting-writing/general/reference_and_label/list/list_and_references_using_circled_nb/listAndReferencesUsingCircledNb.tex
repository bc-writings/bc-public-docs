\documentclass{article}
	\usepackage{tikz}

	\renewcommand\labelenumi{\theenumi}
	\renewcommand{\theenumi}{%
		\protect\begin{tikzpicture}[baseline=(text.base)]%
			\protect\node[
				circle,
				draw,
				color=blue,
				inner sep=1pt
			] (text) {\arabic{enumi}};%
		\protect\end{tikzpicture}%
	}

\end{enumerate}
