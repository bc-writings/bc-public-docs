\documentclass{article}
	\usepackage{amsmath}
	\usepackage{enumitem}
	\usepackage{multicol}

	\SetEnumitemKey{twocol}{
		label=(\Alph{enumi}),
		before=\raggedcolumns\begin{multicols}{2},
		after=\end{multicols}
	}

	\SetEnumitemKey{threecol}{
		label=(\Alph{enumi}),
		before=\raggedcolumns\begin{multicols}{3},
		after=\end{multicols}
	}


\begin{document}

Which fractions are reduced?
\begin{enumerate}[twocol]
	\item $\dfrac{4}{7}$
	\item $\dfrac{8}{24}$
	\item $\dfrac{44}{121}$
	\item $\dfrac{9}{11}$
\end{enumerate}

Which fractions are reduced?

	\item $\dfrac{44}{121}$
	\item $\dfrac{44}{121}$

\end{document}
