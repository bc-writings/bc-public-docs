	\begin{exo}[EB-Suite-Arithm\'etico-G\'eom\'etrique-Fonction-R\'ecurrence-Limite.tex Chapitre 5 : Suites.]
	Soit $(u_n)$ la suite d\'efinie par :\[u_0=1 \textrm{ et pour tout entier naturel } n \textrm{, } u_{n+1}=\dfrac{u_n+8}{2u_n+1}\cdot\] On appelle $f$ la fonction d\'efinie sur $\intervalleoo{-\dfrac{1}{2}}{+\infty}$ par :\[f(x)=\dfrac{x+8}{2x+1}\cdot\]
	\begin{questions}
	\item Etudier les variations de $f(x)$ sur $\intervalleoo{-\dfrac{1}{2}}{+\infty}$.
	\item \label{etude}
	\begin{sousquestions}
	\item Tracer, dans un rep\`ere orthonorm\'e, la courbe $\mathscr{C}_f$ repr\'esentant la fonction $f$ et la droite $\Delta$ d'\'equation $y=x$.
	\item Sur l'axe des abscisses, placer $u_0$, puis construire les termes $u_1$, $u_2$ et $u_3$ en utilisant la courbe $\mathscr{C}_f$ et la droite $\Delta$.
	\item \label{conjectures} Que peut-on conjecturer sur le sens de variation de la suite $(u_n)$ et sur sa limite?
	\end{sousquestions}
	\item soit $(v_n)$ la suite d\'efinie par :\[\textrm{pour tout entier naturel } n \textrm{, }v_n=\dfrac{u_n-2}{u_n+2}\cdot\]
	\begin{sousquestions}
	\item D\'emontrer que $(v_n)$ est une suite g\'eom\'etrique. Calculer le premier terme $v_0$ et la raison $r$.
	\item En d\'eduire l'expression de $v_n$ puis celle de $u_n$ en fonction de $n$.
	\end{sousquestions}
	\item D\'emontrer les conjectures \'emises en \ref{etude} \ref{conjectures}.
	\end{questions}
	\end{exo}
