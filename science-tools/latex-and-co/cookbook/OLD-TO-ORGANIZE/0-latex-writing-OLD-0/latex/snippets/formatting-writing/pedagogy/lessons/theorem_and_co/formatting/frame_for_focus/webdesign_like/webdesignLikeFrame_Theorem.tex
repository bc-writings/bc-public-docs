	\usepackage{environ,tikz}

	\NewEnviron{thm}{%

				\node[
					rectangle,
					rounded corners,
					inner sep=5pt,
					minimum width=\textwidth,
					text=white!93!brown!93!yellow,
					text opacity=1,
					draw=gray,
					ultra thick,
					draw opacity=1,
					fill=black,
					fill opacity=.7
				] (box) {%
					\begin{minipage}{.95\textwidth}\color{white!93!brown!93!yellow}
						\textbf{Theorem}\qquad\BODY
					\end{minipage}%
				};
			\end{tikzpicture}
		\end{center}
	}
