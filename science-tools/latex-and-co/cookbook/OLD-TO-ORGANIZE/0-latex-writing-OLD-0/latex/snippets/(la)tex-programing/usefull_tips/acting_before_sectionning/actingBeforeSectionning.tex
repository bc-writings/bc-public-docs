\documentclass{report}
	\usepackage{etoolbox}

	\pretocmd{\section}{%
		\ifnum\value{section}=0 \else\clearpage\fi
	}{}{}


\begin{document}

\chapter{First Chapter}
\section{Section One One}
\section{Section One Two}
\section{Section One Three}

\end{document}
