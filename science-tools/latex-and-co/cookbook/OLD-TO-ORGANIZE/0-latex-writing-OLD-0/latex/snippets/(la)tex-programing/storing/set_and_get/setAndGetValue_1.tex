\documentclass{article}
	\usepackage{pathkeys}

	\makeatletter
		\newcommand\setvalues[1]{\pathkeys{citronas/var/set:#1}}
		\newcommand\getvalue[1]{\pathkeysval{citronas/var/#1}}
		\def\variablelist{}
		\newcommand\NewVariables[1]{%
			\cptfor#1\dofor{%
				\def\reserved@a####1=####2=####3\@nil{%
					\cptgaddtolist[;]\variablelist{cmd/####1/####2}%
				}%
				\reserved@a##1==\@nil
			}%
			\cptexpanded{\pathkeys{citronas/var/define*:\expandcsonce\variablelist}}%
		}
	\makeatother

	\NewVariables{%
		var1 = default value1,
		var2 = defaultvalue2
	}
	\setvalues{var1=new value1, var2=new value2}
	\edef\x{\getvalue{var1}}


\begin{document}

\getvalue{var1}

\getvalue{var1}

\end{document}
