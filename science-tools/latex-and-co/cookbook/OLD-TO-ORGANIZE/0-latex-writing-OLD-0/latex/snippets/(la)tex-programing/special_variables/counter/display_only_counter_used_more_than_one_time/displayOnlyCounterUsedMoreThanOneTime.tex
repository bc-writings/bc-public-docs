\documentclass{article}
	\makeatletter
		\newcommand{\defineoneshotcounter}[1]{%
			\newcounter{#1}
			\newcounter{#1@helper}
			\expandafter\newcommand\csname show#1\endcsname[1][\arabic]{%
				\stepcounter{#1}%
				\stepcounter{#1@helper}%
				\@ifundefined{#1@reset@\the\csname c@#1@helper\endcsname}{##1{#1}}{}%
			}
			\@namedef{reset#1}{%
				\ifnum\csname c@#1\endcsname=1\if@filesw
				\immediate\write\@auxout{%
					\string\global\string\@namedef{#1@reset@\the\csname
					c@#1@helper\endcsname}{}%
				}%
				\fi\fi
				\setcounter{#1}{0}%
			}
			\AtEndDocument{\@nameuse{reset#1}} % SIC
		}
	\makeatother

	\defineoneshotcounter{FirstCounter}
	\newcommand{\FirstCounterStyle}[1]{\{\alph{#1}\}}

	\defineoneshotcounter{SecundCounter}
	\newcommand{\SecundCounterStyle}[1]{\{\alph{#1}\}} % Probl√®me ?
